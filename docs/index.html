egg-dingtalk-isv

Dingtalk ISV application plugin for Egg.js.

Install

npm i @axolo/egg-dingtalk-isv --save

Usage

// {app_root}/config/plugin.js
exports.dingtalkIsv = {
  enable: true,
  package: '@axolo/egg-dingtalk-isv',
};

Configuration

// {app_root}/config/config.default.js
exports.dingtalkIsv = {
};
